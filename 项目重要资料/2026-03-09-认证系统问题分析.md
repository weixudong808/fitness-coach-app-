### 优先级2：实现体能认证的计算逻辑 🔴

**步骤1：实现 calculateExercisePerformance 函数**
```javascript
const calculateExercisePerformance = async (memberId, requirement) => {
  // 1. 获取会员的有效训练计划
  const { data: plans } = await supabase
    .from('member_plans')
    .select('template_id')
    .eq('member_id', memberId)
    .in('status', ['active', 'completed'])

  // 2. 获取这些计划的所有已完成课次
  const { data: sessions } = await supabase
    .from('training_sessions')
    .select('id')
    .in('template_id', plans.map(p => p.template_id))
    .eq('completed', true)

  // 3. 查询动作记录
  let completedCount = 0
  for (const exercise of requirement.exercises) {
    const { data } = await supabase
      .from('session_exercises')
      .select('reps_standard')
      .in('session_id', sessions.map(s => s.id))
      .eq('exercise_name', exercise.name)
      .limit(1)

    // 4. 检查是否达标
    if (data && data.length > 0) {
      const actual = data[0].reps_standard
      if (meetsRequirement(actual, exercise.requirement)) {
        completedCount++
      }
    }
  }

  return completedCount
}
```

**步骤2：实现 meetsRequirement 函数**
```javascript
const meetsRequirement = (actual, required) => {
  // 解析实际值和要求值
  // 例如：actual = "5分钟", required = "5分钟" → true
  // 例如：actual = "10个", required = "10个" → true
  // 例如：actual = "80kg", required = "80kg" → true

  // 简单实现：字符串匹配
  return actual === required

  // 复杂实现：解析数字和单位，支持大于等于判断
  // 例如：actual = "6分钟" >= required = "5分钟" → true
}
```

**步骤3：修改 calculateSingleProgress**
