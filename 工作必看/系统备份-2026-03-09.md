```sql
-- 添加90条测试记录（从10条增加到100条）
WITH plan AS (
  SELECT template_id FROM member_plans
  WHERE member_id = '2f5278bf-022e-4138-bfc9-6b07af825d31'
  LIMIT 1
),
max_session AS (
  SELECT COALESCE(MAX(session_number), 0) as max_num
  FROM training_sessions
  WHERE template_id = (SELECT template_id FROM plan)
)
INSERT INTO training_sessions (template_id, session_number, training_part, completed, completed_date)
SELECT
  (SELECT template_id FROM plan),
  (SELECT max_num FROM max_session) + row_number() OVER (),
  '测试',
  true,
  CURRENT_DATE - (90 - row_number() OVER ()) * INTERVAL '1 day'
FROM generate_series(1, 90)
RETURNING id;
```

// 统计已点亮的徽章数量
const achievedBadgeCount = computed(() => {
  const activeAchievements = allAchievements.value.filter(a => a.is_active)

// 整体进度

### 3. 性别过滤
```javascript
const filterByGender = (definitions, gender) => {
  return definitions.filter(def => {
    const code = def.code.toLowerCase()

    // 如果 code 包含 _male，只显示给男生
    if (code.includes('_male')) {
      return gender === 'male'
    }

    // 如果 code 包含 _female，只显示给女生
    if (code.includes('_female')) {
      return gender === 'female'
    }

    // 其他认证显示给所有人
    return true
  })
}
```

### 4. 打卡次数计算
```javascript
const calculateCheckInCount = async (memberId) => {
  // 1. 获取会员的有效训练计划
  const { data: memberPlans } = await supabase
    .from('member_plans')
    .select('template_id')
    .eq('member_id', memberId)
    .in('status', ['active', 'completed'])

  if (!memberPlans || memberPlans.length === 0) return 0

  // 2. 统计已完成的课次
  const activeTemplateIds = memberPlans.map(p => p.template_id)
  const { data: sessions } = await supabase
    .from('training_sessions')
    .select('id')
    .in('template_id', activeTemplateIds)
    .eq('completed', true)

  return sessions?.length || 0
}
```
